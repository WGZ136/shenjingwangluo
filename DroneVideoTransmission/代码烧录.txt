一、前视摄像头代码：
1.打开Arduino IDE
2.点击file,点击新建
3.复制粘贴下面的代码：
// ============================================
// ESP32-S3-CAM 前视摄像头代码
// 用于无人机前向拍摄
// ============================================

#include "esp_camera.h"
#include "WiFi.h"

// 20脚 OV2640 摄像头模组 (常见引脚定义)
#define PWDN_GPIO_NUM     -1   // 通常20脚模组不带PWDN控制，填-1
#define RESET_GPIO_NUM    -1   // 通常20脚模组不带RESET控制，填-1
#define XCLK_GPIO_NUM     15   // 外部时钟引脚
#define SIOD_GPIO_NUM      4   // SCCB/I2C 数据引脚
#define SIOC_GPIO_NUM      5   // SCCB/I2C 时钟引脚
#define Y9_GPIO_NUM       16   // D7 (MSB)
#define Y8_GPIO_NUM       17   // D6
#define Y7_GPIO_NUM       18   // D5
#define Y6_GPIO_NUM       12   // D4
#define Y5_GPIO_NUM       10   // D3
#define Y4_GPIO_NUM        8   // D2
#define Y3_GPIO_NUM        9   // D1
#define Y2_GPIO_NUM       11   // D0 (LSB)
#define VSYNC_GPIO_NUM     6   // 垂直同步
#define HREF_GPIO_NUM      7   // 水平参考/同步
#define PCLK_GPIO_NUM     13   // 像素时钟

// WiFi设置
const char* ssid = "Drone_Cam_Front";  // 前视摄像头WiFi名称
const char* password = "12345678";     // WiFi密码

WiFiServer server(80);

void setup() {
  Serial.begin(115200);
  Serial.println("\n=== 无人机前视摄像头启动 ===");
  
  // 初始化摄像头
  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;
  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;
  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;
  config.pin_sscb_sda = SIOD_GPIO_NUM;
  config.pin_sscb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;
  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_JPEG;
  
  // 图像质量设置
  config.frame_size = FRAMESIZE_QVGA;  // 320x240，适合图传
  config.jpeg_quality = 12;            // 0-63，越小越清晰
  config.fb_count = 2;
  
  // 初始化摄像头
  esp_err_t err = esp_camera_init(&config);
  if (err != ESP_OK) {
    Serial.printf("摄像头初始化失败: 0x%x", err);
    return;
  }
  Serial.println("✅ 摄像头初始化成功");
  
  // 创建WiFi热点
  Serial.println("📡 创建WiFi热点...");
  WiFi.softAP(ssid, password);
  
  IPAddress IP = WiFi.softAPIP();
  Serial.print("✅ WiFi热点创建成功: ");
  Serial.println(IP);
  Serial.print("📶 连接密码: ");
  Serial.println(password);
  
  // 启动Web服务器
  server.begin();
  Serial.println("🌐 Web服务器已启动");
}

void loop() {
  WiFiClient client = server.available();
  
  if (client) {
    Serial.println("📱 新客户端连接");
    
    // 读取HTTP请求
    String request = client.readStringUntil('\r');
    client.readStringUntil('\n');
    
    // 如果是获取图像请求
    if (request.indexOf("GET /capture") >= 0) {
      camera_fb_t * fb = esp_camera_fb_get();
      if (!fb) {
        Serial.println("摄像头捕获失败");
        client.println("HTTP/1.1 500 Internal Server Error");
        client.println();
        return;
      }
      
      // 发送HTTP响应头
      client.println("HTTP/1.1 200 OK");
      client.println("Content-Type: image/jpeg");
      client.println("Content-Length: " + String(fb->len));
      client.println("Access-Control-Allow-Origin: *");
      client.println();
      
      // 发送图像数据
      client.write(fb->buf, fb->len);
      esp_camera_fb_return(fb);
      
      Serial.println("📸 图像已发送");
    } 
    // 如果是视频流请求
    else if (request.indexOf("GET /stream") >= 0) {
      client.println("HTTP/1.1 200 OK");
      client.println("Content-Type: multipart/x-mixed-replace; boundary=frame");
      client.println("Access-Control-Allow-Origin: *");
      client.println();
      
      while (client.connected()) {
        camera_fb_t * fb = esp_camera_fb_get();
        if (!fb) {
          Serial.println("摄像头捕获失败");
          break;
        }
        
        client.println("--frame");
        client.println("Content-Type: image/jpeg");
        client.println("Content-Length: " + String(fb->len));
        client.println();
        client.write(fb->buf, fb->len);
        client.println();
        
        esp_camera_fb_return(fb);
        delay(100);  // 控制帧率
      }
    }
    // 默认页面
    else {
      client.println("HTTP/1.1 200 OK");
      client.println("Content-Type: text/html");
      client.println();
      client.println("<!DOCTYPE html><html><head><title>无人机前视摄像头</title></head><body>");
      client.println("<h1>无人机前视摄像头</h1>");
      client.println("<p><a href='/capture'>拍摄单张照片</a></p>");
      client.println("<p><a href='/stream'>实时视频流</a></p>");
      client.println("</body></html>");
    }
    
    client.stop();
    Serial.println("客户端断开连接");
  }
}
